-- Users & Auth
CREATE TABLE users (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  email TEXT UNIQUE NOT NULL,           -- AES-256-GCM encrypted
  email_hash TEXT UNIQUE NOT NULL,      -- SHA-256, for lookups
  password_hash TEXT NOT NULL,          -- argon2id
  totp_secret TEXT,                     -- encrypted
  is_2fa_enabled BOOLEAN DEFAULT false,
  role TEXT DEFAULT 'ANALYST',
  is_active BOOLEAN DEFAULT true,
  email_verified_at TIMESTAMPTZ,
  last_login_at TIMESTAMPTZ,
  failed_login_attempts INT DEFAULT 0,
  locked_until TIMESTAMPTZ,
  created_at TIMESTAMPTZ DEFAULT NOW()
);

CREATE TABLE refresh_tokens (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id UUID REFERENCES users(id) ON DELETE CASCADE,
  token_hash TEXT UNIQUE NOT NULL,      -- SHA-256 of token
  family_id UUID NOT NULL,             -- for reuse detection
  expires_at TIMESTAMPTZ NOT NULL,
  revoked_at TIMESTAMPTZ,
  created_at TIMESTAMPTZ DEFAULT NOW()
);

CREATE TABLE backup_codes (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id UUID REFERENCES users(id) ON DELETE CASCADE,
  code_hash TEXT NOT NULL,             -- bcrypt
  used_at TIMESTAMPTZ
);

SECTION 8: WEBSOCKET ARCHITECTURE
════════════════════════════════════════════════════════════════════════════════

Server (FastAPI + Redis Pub/Sub):

class ConnectionManager:
    - Per-user connection registry (dict[user_id, list[WebSocket]])
    - Per-symbol subscription registry (dict[symbol, set[user_id]])
    - Heartbeat: ping every 30s, close if no pong in 10s
    - Reconnection: client implements exponential backoff (1s → 2s → 4s → max 30s)
    - Max 10 concurrent connections per user
    - Auth: JWT token passed as query param on connect (ws://.../ws?token=...)
    
Message Types (JSON with "type" discriminator):
  Server → Client:
  - {type: "tick", symbol, price, volume, change_pct, timestamp}
  - {type: "prediction_update", symbol, model, direction, confidence}
  - {type: "alert_triggered", alert_id, message, severity, symbol}
  - {type: "regime_change", symbol, old_regime, new_regime, confidence}
  - {type: "anomaly_detected", symbol, anomaly_type, severity, description}
  - {type: "portfolio_update", pnl, pnl_pct, positions}
  - {type: "heartbeat", server_time}

  Client → Server:
  - {type: "subscribe", symbols: ["RELIANCE.NS", "BTC-USD"]}
  - {type: "unsubscribe", symbols: [...]}
  - {type: "pong"}

Client Hook (React):
  useWebSocket(symbols: string[]) returns:
  - ticks: Map<symbol, TickData>
  - predictions: Map<symbol, PredictionData>
  - alerts: AlertEvent[]
  - connectionStatus: "connected" | "reconnecting" | "disconnected"
  - anomalies: AnomalyEvent[]
